Part 1. Background

Comparing two treatments (A and B) for acute respiratory infection. Both treatment aims to cure patients, but they differ in: -Upfront treatment cost -Cure rate -Risk of adverse drug reaction (ADRs) -associated downstream costs

key considerations: -ARDs require additional management costs -Treatment failure require additional treatment cost -Health outcomes differ between cured and uncured patients

Task: -Determine the incremental cost-effectiveness ratio (ICER) -Decide whether Treatment A represents good value for money compare to Treatment B

What we know:

Probabilities: -Treatment A adverse reaction: 0.2 No adverse reaction: 0.8 Cured: 0.9 Not cured: 0.1 -Treatment B adverse reaction: 0.3 No adverse reaction: 0.7 Cured: 0.8 Not cured: 0.2

Costs: -Treatment A: $100 -Treatment B: $80 -Adverse reaction: $500 -Not cured: $150

Life years: -Cured:20 -Not cured:10

Part 4. Build the decision tree model

# We wrap the decision tree calculation into a funstion so we can: # -easily run the model with different parameters # -perform sensitivity analysis # -keep code organized and reusable

dec_tree <- function(params) {
  with(as.list(params),{
    

#-----------------------------------------------------------------------------
### Step 1: Calculate probabilities 
#-----------------------------------------------------------------------------
    
 # For each treatment, we have 4 mutually exclusive pathway based on:
 # -whether patients experience ADR (yes/no)
 # -whether patient is cured (yes/no)

 # Treatment A pathways

ep1 <- (1 - p.TxA_ADR) * p.TxA_Cured           #pathway 1: No ADR, Cured
ep2 <- (1 - p.TxA_ADR) * (1 - p.TxA_Cured)     #pathway 2: No ADR, Not Cured
ep3 <- p.TxA_ADR * p.TxA_Cured                 #pathway 3: ADR, Cured
ep4 <- p.TxA_ADR * (1 - p.TxA_Cured)           #pathway 4: ADR, Not Cured


 #Treatment B pathways

ep5 <- (1 - p.TxB_ADR) * p.TxB_Cured           #pathway 5: No ADR, Cured
ep6 <- (1 - p.TxB_ADR) * (1 - p.TxB_Cured)     #pathway 6: No ADR, Not Cured
ep7 <- p.TxB_ADR * p.TxB_Cured                 #pathway 7: ADR, Cured
ep8 <- p.TxB_ADR * (1 - p.TxB_Cured)           #pathway 8: ADR, Not Cured


#-----------------------------------------------------------------------------
### Step 2: Calculate total cost per pathway
#-----------------------------------------------------------------------------

 # Treatment A pathway costs

tc1 <- c.TxA + c.NoADR + c.Cured                #pathway 1 cost
tc2 <- c.TxA + c.NoADR + c.NotCured             #pathway 2 cost
tc3 <- c.TxA + c.ADR + c.Cured                  #pathway 3 cost
tc4 <- c.TxA + c.ADR + c.NotCured               #pathway 4 cost


 # Treatment B pathway costs

tc5 <- c.TxB + c.NoADR + c.Cured                #pathway 5 cost
tc6 <- c.TxB + c.NoADR + c.NotCured             #pathway 6 cost
tc7 <- c.TxB + c.ADR + c.Cured                  #pathway 7 cost
tc8 <- c.TxB + c.ADR + c.NotCured               #pathway 8 cost

#-----------------------------------------------------------------------------
### Step 3: Calculate expected values (probabilities - weighted outcomes)
#-----------------------------------------------------------------------------

 # Expected value = sum of (probabilities * outcomes) for all pathways
 # This is core of decision tree analysis

 # Expected total costs (sum of weighted pathway costs)

etc.TxA <- (ep1 * tc1) + (ep2 * tc2) + (ep3 * tc3) + (ep4 * tc4) 
etc.TxB <- (ep5 * tc5) + (ep6 * tc6) + (ep7 * tc7) + (ep8 * tc8) 


 # Expected life years (sum of weighted pathway life years)

ely.TxA <- (ep1 * ly.Cured) + (ep2 * ly.NotCured) + (ep3 * ly.Cured) + (ep4 * ly.NotCured) 
ely.TxB <- (ep5 * ly.Cured) + (ep6 * ly.NotCured) + (ep7 * ly.Cured) + (ep8 * ly.NotCured) 

#-----------------------------------------------------------------------------
### Step 4: Calculate incremental cost and ICER
#-----------------------------------------------------------------------------

 # Incremental values (Treatment A vs. Treatment B)

IC <- etc.TxA - etc.TxB
IE <- ely.TxA - ely.TxB
ICER = IC / IE

#-----------------------------------------------------------------------------
### Step 5: Prepare output
#-----------------------------------------------------------------------------

 # Combine all results

C <- c(etc.TxA, etc.TxB)
LY <- c(ely.TxA, ely.TxB)


 # Add names for clarity

names(C) <- paste("C", c("TxA", "TxB"), sep = "_")
names(LY) <- paste("LY", c("TxA", "TxB"), sep = "_")
names(IC) <- "Incr Costs"
names(IE) <- "Incr Life Years"
names(ICER) <- "ICER"


 # Return all results
 
return(c(C, LY, IC, IE, ICER)) 

 })
}

Part 6. One way sensitivity analysis

# A one-way sensitivity analysis evaluates the impact of parameter uncertainty on the decision model results. Changing one parameter’s values across a specific range allows decision modelers to investigate its impact on the output. For example, if we are interested in the impact of changing the Probability of Cured for Treatment A on the Total Costs, we can vary this probablity across a range of 0% to 100%.

# Define the range for probability of cure with Treatment A
# We'll test 11 values evenly spaced from 0.0 to 1.0
p.TxA_Cured_range <- seq(0.00, 1.00, length.out = 11)

# Create input matrix with varied parameter
m.owsa.input <- data.frame(

# Tornado diagrams show which parameters have the greatest impact on our results. # We vary each parameter across a plausible range and measure the effect on ICER.

# Define parameter ranges (typically ±25% or based on literature)
param_ranges <- list(
  p.TxA_ADR    = c(base = 0.20,  low = 0.15,  high = 0.25),
  p.TxB_ADR    = c(base = 0.30,  low = 0.225, high = 0.375),
  ly.Cured     = c(base = 20,    low = 15,    high = 25),
  ly.NotCured  = c(base = 10,    low = 7.5,   high = 12.5),
  c.TxA        = c(base = 100,   low = 75,    high = 125),
  c.TxB        = c(base = 80,    low = 60,    high = 100),
  c.ADR        = c(base = 500,   low = 375,   high = 625),
  c.NotCured   = c(base = 150,   low = 113,   high = 188)
)

# Create parameter name vector for later use
paramNames <- names(param_ranges)

# Initialize list to store input scenarios
l.tor.in <- vector("list", length(param_ranges))
names(l.tor.in) <- paramNames

# For each parameter, create 3 scenarios (base, low, high)
for (i in seq_along(param_ranges)) {
  param_name <- paramNames[i]
  
  # Create 3 rows with base case values
  temp_df <- input[rep(1, 3), ]
  rownames(temp_df) <- c("base", "low", "high")
  
  # Replace the current parameter with its range
  temp_df[[param_name]] <- param_ranges[[i]]
  
  # Store in list
  l.tor.in[[i]] <- temp_df
}

# Initialize list to store outputs
l.tor.out <- vector("list", length(param_ranges))
names(l.tor.out) <- paramNames

# Run model for each parameter's range and extract ICER
for (i in 1:length(param_ranges)) {
  l.tor.out[[i]] <- t(apply(l.tor.in[[i]], 1, dec_tree))[, 7]  # Column 7 is ICER
}

# Create matrix for tornado plot
m.tor <- matrix(
  unlist(l.tor.out),
  nrow = length(param_ranges),
  ncol = 3,
  byrow = TRUE,
  dimnames = list(paramNames, c("basecase", "low", "high"))
)


# Define readable parameter labels
param_labels <- c(
  "P(ADR | Treatment A)",
  "P(ADR | Treatment B)",
  "Life Years (Cured)",
  "Life Years (Not Cured)",
  "Cost of Treatment A",
  "Cost of Treatment B",
  "Cost of ADR",
  "Cost of Not Cured"
)

# Calculate impact (range) for ranking
ymean <- m.tor[1, 1]  # Base case ICER
ySize <- abs(m.tor[, 3] - m.tor[, 2])  # Range (high - low)

# Rank by impact
rankY <- order(ySize)
nParams <- length(param_labels)

# Create data frame for plotting - one row per parameter
tornado_data <- data.frame(
  Parameter = param_labels[rankY],
  Low_ICER = m.tor[rankY, 2],
  High_ICER = m.tor[rankY, 3],
  Base_ICER = m.tor[rankY, 1]
)

# Convert Parameter to ordered factor
tornado_data$Parameter <- factor(tornado_data$Parameter, levels = param_labels[rankY])

# Calculate axis range with padding
x_min <- min(c(tornado_data$Low_ICER, tornado_data$High_ICER))
x_max <- max(c(tornado_data$Low_ICER, tornado_data$High_ICER))
x_range <- x_max - x_min
x_lower <- x_min - x_range * 0.20  # Extra space on left for labels
x_upper <- x_max + x_range * 0.20  # Extra space on right for labels

# Offset for labels - position them just outside the bar ends
label_offset <- x_range * 0.02

# Create beautiful tornado plot with custom colors
tornado_plot <- ggplot(tornado_data, aes(y = Parameter)) +
  # Add bars as line segments with thickness
  geom_segment(
    aes(x = Low_ICER, xend = Base_ICER, yend = Parameter),
    color = "blue",
    linewidth = 8,
    alpha = 0.85
  ) +
  geom_segment(
    aes(x = Base_ICER, xend = High_ICER, yend = Parameter),
    color = "green",
    linewidth = 8,
    alpha = 0.85
  ) +
  
  # Add ICER value labels positioned at actual ICER values
  # Low parameter value labels - position based on whether Low_ICER is left or right of base
  geom_text(
    aes(
      x = ifelse(Low_ICER < Base_ICER, Low_ICER - label_offset, Low_ICER + label_offset),
      label = sprintf("%.1f", Low_ICER),
      hjust = ifelse(Low_ICER < Base_ICER, 1, 0)  # Right-align if on left, left-align if on right
    ),
    size = 3.2,
    color = "darkgrey",
    fontface = "bold"
  ) +
  # High parameter value labels - position based on whether High_ICER is left or right of base
  geom_text(
    aes(
      x = ifelse(High_ICER < Base_ICER, High_ICER - label_offset, High_ICER + label_offset),
      label = sprintf("%.1f", High_ICER),
      hjust = ifelse(High_ICER < Base_ICER, 1, 0)  # Right-align if on left, left-align if on right
    ),
    size = 3.2,
    color = "darkgrey",
    fontface = "bold"
  ) +
  
  # Set x-axis limits with extra space for legend and labels
  scale_x_continuous(
    limits = c(x_lower, x_upper),
    expand = c(0, 0)
  ) +
  
  # Remove extra spacing - normal discrete scale
  scale_y_discrete() +
  
  # Base case reference line at actual ICER value
  geom_vline(xintercept = ymean, linetype = "solid", linewidth = 1, color = "black") +
  
  # Labels - include base case ICER in the y-axis title
  labs(
    title = "Tornado Diagram: Parameter Uncertainty Impact on ICER",
    subtitle = sprintf("Bars show ICER range when parameters vary (Base Case ICER: £%.1f - shown by black line)", ymean),
    x = "ICER (£ per Life Year Gained)",
    y = "Parameters (Ordered by Impact)"
  ) +
  
  # Clean theme with custom styling
  theme_minimal(base_size = 13) +
  theme(
    plot.title = element_text(face = "bold", size = 16, hjust = 0, color = "black"),
    plot.subtitle = element_text(size = 11, hjust = 0, color = "darkgrey", margin = margin(b = 15)),
    axis.title = element_text(face = "bold", size = 12, color = "darkgrey"),
    axis.text = element_text(size = 11, color = "darkgrey"),
    axis.text.y = element_text(size = 11),
    panel.grid.major.y = element_blank(),
    panel.grid.major.x = element_line(color = "gray90"),
    panel.grid.minor = element_blank(),
    plot.margin = margin(20, 20, 20, 20),
    plot.title.position = "plot"
  ) +
  
  # Add manual legend in the plot area (left side)
  annotate(
    "segment",
    x = x_min - x_range * 0.15,
    xend = x_min - x_range * 0.10,
    y = 1.2,
    yend = 1.2,
    color = "blue",
    linewidth = 5
  ) +
  annotate(
    "text",
    x = x_min - x_range * 0.08,
    y = 1.2,
    label = "Low Parameter Value",
    hjust = 0,
    size = 3.5,
    color = "darkgrey"
  ) +
  annotate(
    "segment",
    x = x_min - x_range * 0.15,
    xend = x_min - x_range * 0.10,
    y = 0.6,

    y = 0.6,

    hjust = 0,

    color = "darkgrey"
  )
